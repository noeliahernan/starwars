CREATE TABLE FILMS (
ID_FILM NUMBER PRIMARY KEY NOT NULL,
TITLE VARCHAR,
EPISODE_ID NUMBER,
OPENING_CRAWL VARCHAR,
DIRECTOR VARCHAR,
PRODUCER VARCHAR,
RELEASE_DATE DATE,
CREATED DATE,
EDITED DATE,
URL VARCHAR);

CREATE TABLE PEOPLE(
ID_PERSON NUMBER PRIMARY KEY NOT NULL,
NAME VARCHAR,
HEIGHT NUMBER,
MASS NUMBER,
HAIR_COLOR VARCHAR,
SKIN_COLOR VARCHAR,
EYE_COLOR VARCHAR,
BIRTH_YEAR VARCHAR,
GENDER VARCHAR,
HOMEWORLD VARCHAR,
CREATED DATE,
EDITED DATE,
URL VARCHAR);	


CREATE TABLE STARSHIPS(ID_STARSHIP NUMBER PRIMARY KEY NOT NULL,
NAME VARCHAR,
MODEL VARCHAR,
MANUFACTURER VARCHAR,
COST_IN_CREDITS VARCHAR,
LENGTH VARCHAR,
MAX_ATMOSPHERING_SPEED VARCHAR,
CREW VARCHAR,
PASSENGERS VARCHAR,
CARGO_CAPACITY VARCHAR,
CONSUMABLES VARCHAR,
HYPERDRIVE_RATING VARCHAR,
MGLT VARCHAR,
STARSHIP_CLASS VARCHAR,
CREATED DATE,
EDITED DATE,
URL VARCHAR);


CREATE TABLE FILMS_PEOPLE (
ID_FILM_PERSON NUMBER PRIMARY KEY NOT NULL,
ID_FILM NUMBER NOT NULL,
ID_PERSON NUMBER NOT NULL);

ALTER TABLE FILMS_PEOPLE
    ADD FOREIGN KEY (ID_PERSON) 
    REFERENCES PEOPLE(ID_PERSON);

ALTER TABLE FILMS_PEOPLE
    ADD FOREIGN KEY (ID_FILM) 
    REFERENCES FILMS(ID_FILM);
	
	
	CREATE TABLE PEOPLE_STARSHIPS (
ID_PERSON_STARSHIP NUMBER PRIMARY KEY NOT NULL,
ID_PERSON NUMBER NOT NULL,
ID_STARSHIP NUMBER NOT NULL);

ALTER TABLE PEOPLE_STARSHIPS
    ADD FOREIGN KEY (ID_STARSHIP) 
    REFERENCES STARSHIPS(ID_STARSHIP);

ALTER TABLE PEOPLE_STARSHIPS
    ADD FOREIGN KEY (ID_PERSON) 
    REFERENCES PEOPLE(ID_PERSON);


CREATE TABLE FILMS_STARSHIPS (
ID_FILM_STARSHIP NUMBER PRIMARY KEY NOT NULL,
ID_FILM NUMBER NOT NULL,
ID_STARSHIP NUMBER NOT NULL);

ALTER TABLE FILMS_STARSHIPS
    ADD FOREIGN KEY (ID_STARSHIP) 
    REFERENCES STARSHIPS(ID_STARSHIP);

ALTER TABLE FILMS_STARSHIPS
    ADD FOREIGN KEY (ID_FILM) 
    REFERENCES FILMS(ID_FILM);


	CREATE SEQUENCE SEQ_FILMS INCREMENT by 1;
